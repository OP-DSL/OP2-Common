project(OP2-Jac1)

# Require CMake 2.8
cmake_minimum_required(VERSION 2.8)

include(../common.cmake)

foreach(VARIANT ${OP2_BUILD_VARIANTS})

  set(KERNEL_HEADERS ${VARIANT}/res.h ${VARIANT}/update.h)

  # simple sequential version
  op2_application(jac1_${VARIANT}_seq LIBS op2_seq SOURCES ${VARIANT}/jac.cpp)

  # x86 version using kernel files generated by op2.m
  op2_application(jac1_${VARIANT}_openmp LIBS op2_openmp
    SOURCES ${VARIANT}/jac_op.cpp ${VARIANT}/openmp/jac_kernels.cpp ${KERNEL_HEADERS})

  # CUDA version using kernel files generated by op2.m
  op2_application(jac1_${VARIANT}_cuda LIBS op2_cuda
    SOURCES ${VARIANT}/jac_op.cpp ${VARIANT}/cuda/jac_kernels.cu ${KERNEL_HEADERS})

  # MPI with sequential-nodes version
  op2_application(jac1_${VARIANT}_mpi LIBS op2_mpi
    SOURCES ${VARIANT}/jac_mpi.cpp ${KERNEL_HEADERS})

endforeach()
