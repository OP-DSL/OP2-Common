project(OP2-Airfoil-plain)

# Require CMake 2.8
cmake_minimum_required(VERSION 2.8)

include(../../common.cmake)
include(GenerateMesh)
generate_mesh(AIRFOIL new_grid.dat naca0012)

foreach(VARIANT ${OP2_BUILD_VARIANTS})

  set(KERNEL_HEADERS ${VARIANT}/save_soln.h ${VARIANT}/adt_calc.h ${VARIANT}/res_calc.h ${VARIANT}/bres_calc.h)

  # simple sequential version
  op2_application(airfoil_${VARIANT}_seq DEPENDS AIRFOIL_grid LIBS op2_seq
    SOURCES ${VARIANT}/airfoil.cpp ${KERNEL_HEADERS})

  # x86 version using kernel files generated by op2.m
  op2_application(airfoil_${VARIANT}_openmp DEPENDS AIRFOIL_grid LIBS op2_openmp
    SOURCES ${VARIANT}/airfoil_op.cpp ${VARIANT}/airfoil_kernels.cpp ${KERNEL_HEADERS})

  # CUDA version using kernel files generated by op2.m
  op2_application(airfoil_${VARIANT}_cuda DEPENDS AIRFOIL_grid LIBS op2_cuda
    SOURCES ${VARIANT}/airfoil_op.cpp ${VARIANT}/airfoil_kernels.cu ${KERNEL_HEADERS})

  # MPI with sequential-nodes version
  op2_application(airfoil_${VARIANT}_mpi DEPENDS AIRFOIL_grid LIBS op2_mpi
    SOURCES ${VARIANT}/airfoil_mpi.cpp ${KERNEL_HEADERS})

  # MPI + CUDA version
  op2_application(airfoil_${VARIANT}_mpi_cuda DEPENDS AIRFOIL_grid LIBS op2_mpi_cuda
    SOURCES ${VARIANT}/airfoil_mpi_op.cpp ${VARIANT}/airfoil_kernels.cu
    ${VARIANT}/save_soln.h ${VARIANT}/adt_calc.h ${VARIANT}/res_calc.h ${VARIANT}/bres_calc.h ${VARIANT}/update.h)

  #MPI + OpenMP version
  op2_application(airfoil_${VARIANT}_mpi_openmp DEPENDS AIRFOIL_grid LIBS op2_mpi op2_openmp
    SOURCES ${VARIANT}/airfoil_mpi_op.cpp ${VARIANT}/airfoil_kernels.cpp
    ${VARIANT}/save_soln.h ${VARIANT}/adt_calc.h ${VARIANT}/res_calc.h ${VARIANT}/bres_calc.h ${VARIANT}/update.h)

endforeach()
