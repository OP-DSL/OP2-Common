project(OP2-Airfoil-HDF5)

# Require CMake 2.8
cmake_minimum_required(VERSION 2.8)

include(../../common.cmake)
include(GenerateMesh)
generate_mesh(AIRFOIL new_grid.dat naca0012)

# All these require HDF5
if (TARGET op2_hdf5 AND OP2_BUILD_DP)

  set(KERNEL_HEADERS dp/save_soln.h dp/adt_calc.h dp/res_calc.h dp/bres_calc.h)

  # HDF5 mesh conversion utility
  op2_application(convert_mesh LIBS op2_mpi SOURCES dp/convert_mesh.cpp)

  generate_hdf5_mesh(AIRFOIL new_grid convert_mesh)

  # simple sequential version
  op2_application(airfoil_hdf5_dp_seq DEPENDS AIRFOIL_h5_grid LIBS op2_seq op2_hdf5
    SOURCES dp/airfoil.cpp)

  # x86 version using kernel files generated by op2.m
  op2_application(airfoil_hdf5_dp_openmp DEPENDS AIRFOIL_h5_grid LIBS op2_openmp op2_hdf5
    SOURCES dp/airfoil_op.cpp dp/airfoil_kernels.cpp ${KERNEL_HEADERS})

  # CUDA version using kernel files generated by op2.m
  op2_application(airfoil_hdf5_dp_cuda DEPENDS AIRFOIL_h5_grid LIBS op2_cuda op2_hdf5
    SOURCES dp/airfoil_op.cpp dp/airfoil_kernels.cu ${KERNEL_HEADERS})

  # MPI with sequential-nodes version
  op2_application(airfoil_hdf5_dp_mpi DEPENDS AIRFOIL_h5_grid LIBS op2_mpi
    SOURCES dp/airfoil.cpp ${KERNEL_HEADERS})

  # MPI + CUDA version
  op2_application(airfoil_hdf5_dp_mpi_cuda DEPENDS AIRFOIL_grid LIBS op2_mpi_cuda
    SOURCES dp/airfoil_op.cpp dp/airfoil_kernels.cu
    dp/save_soln.h dp/adt_calc.h dp/res_calc.h dp/bres_calc.h dp/update.h)

  #MPI + OpenMP version
  op2_application(airfoil_hdf5_dp_mpi_openmp DEPENDS AIRFOIL_grid LIBS op2_mpi op2_openmp
    SOURCES dp/airfoil_op.cpp dp/airfoil_kernels.cpp
    dp/save_soln.h dp/adt_calc.h dp/res_calc.h dp/bres_calc.h dp/update.h)

endif()
